**Change 1 — Add @DataClassName to ALL 5 table classes:**

Add `@DataClassName('XxxRow')` annotation above each table class. This renames the Drift-generated data classes to avoid conflicts with Freezed domain models:

```dart
@DataClassName('ReminderChainRow')
class ReminderChains extends Table {
  // ... existing column definitions unchanged
}

@DataClassName('ReminderRow')
class Reminders extends Table {
  // ... existing column definitions unchanged
}

@DataClassName('ChainEdgeRow')
class ChainEdges extends Table {
  // ... existing column definitions unchanged
}

@DataClassName('ConfirmationRow')
class Confirmations extends Table {
  // ... existing column definitions unchanged
}

@DataClassName('MealAnchorRow')
class MealAnchors extends Table {
  // ... existing column definitions unchanged
}
```

**WHY:** Without this, Drift generates `ReminderChain`, `Reminder`, `ChainEdge`, `Confirmation`, `MealAnchor` — identical names to our Freezed domain models. Any file importing both (repositories) would have unresolvable name conflicts. The `Row` suffix convention clearly distinguishes "database row" from "domain model."

**chain_dao.dart** — Handles `reminder_chains` and `chain_edges` tables. Replace the TODO placeholder:

```dart
import 'package:drift/drift.dart';

import '../../../core/database/app_database.dart';

part 'chain_dao.g.dart';

/// Data access object for reminder chains and their DAG edges.
///
/// Provides reactive streams ([watchAllChains], [watchEdgesForChain]) and
/// CRUD operations for the chain aggregate.
@DriftAccessor(tables: [ReminderChains, ChainEdges])
class ChainDao extends DatabaseAccessor<AppDatabase> with _$ChainDaoMixin {
  ChainDao(super.db);

  // --------------- Chains ---------------

  /// Watch all chains ordered by creation date (newest first).
  Stream<List<ReminderChainRow>> watchAllChains() {
    return (select(reminderChains)
          ..orderBy([(c) => OrderingTerm.desc(c.createdAt)]))
        .watch();
  }

  /// Watch a single chain by ID. Emits null if not found.
  Stream<ReminderChainRow?> watchChainById(int id) {
    return (select(reminderChains)..where((c) => c.id.equals(id)))
        .watchSingleOrNull();
  }

  /// Insert a new chain. Returns the auto-generated ID.
  Future<int> insertChain(ReminderChainsCompanion companion) {
    return into(reminderChains).insert(companion);
  }

  /// Update an existing chain. Returns true if a row was updated.
  Future<bool> updateChain(ReminderChainsCompanion companion) {
    return update(reminderChains).replace(companion);
  }

  /// Delete a chain by ID. Returns the number of deleted rows.
  /// Note: Also delete associated edges and reminders via the repository.
  Future<int> deleteChain(int id) {
    return (delete(reminderChains)..where((c) => c.id.equals(id))).go();
  }

  // --------------- Edges ---------------

  /// Watch all edges for a specific chain.
  Stream<List<ChainEdgeRow>> watchEdgesForChain(int chainId) {
    return (select(chainEdges)..where((e) => e.chainId.equals(chainId)))
        .watch();
  }

  /// Insert a new edge. Returns the auto-generated ID.
  Future<int> insertEdge(ChainEdgesCompanion companion) {
    return into(chainEdges).insert(companion);
  }

  /// Delete a single edge by ID.
  Future<int> deleteEdge(int id) {
    return (delete(chainEdges)..where((e) => e.id.equals(id))).go();
  }

  /// Delete all edges for a chain. Used when deleting a chain.
  Future<int> deleteEdgesForChain(int chainId) {
    return (delete(chainEdges)..where((e) => e.chainId.equals(chainId))).go();
  }
}
```

```dart
/// Data access object for individual reminders within chains.
///
/// Provides reactive streams for active reminders and per-chain queries,
/// plus standard CRUD operations.
@DriftAccessor(tables: [Reminders])
class ReminderDao extends DatabaseAccessor<AppDatabase>
    with _$ReminderDaoMixin {
  ReminderDao(super.db);

  /// Watch all active reminders, ordered by scheduled time (earliest first).
  /// Reminders with null scheduledAt sort last (not yet resolved by anchor).
  Stream<List<ReminderRow>> watchActiveReminders() {
    return (select(reminders)
          ..where((r) => r.isActive.equals(true))
          ..orderBy([
            (r) => OrderingTerm.asc(
                  r.scheduledAt,
                  nulls: NullsOrder.last,
                ),
          ]))
        .watch();
  }

  /// Watch all reminders belonging to a specific chain.
```

**confirmation_dao.dart** — Handles `confirmations` table. Replace the TODO placeholder:

```dart
import 'package:drift/drift.dart';

import '../../../core/database/app_database.dart';

part 'confirmation_dao.g.dart';

/// Data access object for reminder confirmations (done/snoozed/skipped).
///
/// Confirmations are append-only — a new row is inserted for each action.
/// The latest confirmation for a reminder determines its current state.
@DriftAccessor(tables: [Confirmations])
class ConfirmationDao extends DatabaseAccessor<AppDatabase>
    with _$ConfirmationDaoMixin {
  ConfirmationDao(super.db);

  /// Watch all confirmations for a specific reminder, newest first.
  Stream<List<ConfirmationRow>> watchConfirmationsForReminder(int reminderId) {
    return (select(confirmations)
          ..where((c) => c.reminderId.equals(reminderId))
          ..orderBy([(c) => OrderingTerm.desc(c.confirmedAt)]))
        .watch();
  }

  /// Watch the most recent confirmation for a reminder.
  /// Returns null if no confirmation exists yet.
  Stream<ConfirmationRow?> watchLatestConfirmation(int reminderId) {
    return (select(confirmations)
          ..where((c) => c.reminderId.equals(reminderId))
          ..orderBy([(c) => OrderingTerm.desc(c.confirmedAt)])
          ..limit(1))
        .watchSingleOrNull();
  }

  /// Insert a new confirmation record. Returns the auto-generated ID.
  /// Confirmations are append-only — do NOT update or delete.
  Future<int> insertConfirmation(ConfirmationsCompanion companion) {
    return into(confirmations).insert(companion);
  }
}
```

**anchor_dao.dart** — Handles `meal_anchors` table. This is a NEW file (not in Phase 01 scaffold). Create it:

```dart
import 'package:drift/drift.dart';

import '../../../core/database/app_database.dart';

part 'anchor_dao.g.dart';

/// Data access object for meal time anchors (breakfast, lunch, dinner).
///
/// Each user has 3 anchors with configurable default times.
/// When a meal is confirmed, [confirmedAt] is set and dependent
/// reminders are recalculated by the AnchorResolver.
@DriftAccessor(tables: [MealAnchors])
class AnchorDao extends DatabaseAccessor<AppDatabase>
    with _$AnchorDaoMixin {
  AnchorDao(super.db);

  /// Watch all meal anchors.
  Stream<List<MealAnchorRow>> watchAllAnchors() {
    return select(mealAnchors).watch();
  }

  /// Watch a specific anchor by meal type ('breakfast', 'lunch', 'dinner').
  Stream<MealAnchorRow?> watchAnchorByMealType(String mealType) {
    return (select(mealAnchors)
          ..where((a) => a.mealType.equals(mealType)))
        .watchSingleOrNull();
  }

  /// Insert a new meal anchor. Returns the auto-generated ID.
  Future<int> insertAnchor(MealAnchorsCompanion companion) {
    return into(mealAnchors).insert(companion);
  }

  /// Update an existing anchor (e.g., change default time or set confirmedAt).
  Future<bool> updateAnchor(MealAnchorsCompanion companion) {
    return update(mealAnchors).replace(companion);
  }

  /// Upsert an anchor — insert if not exists, update if exists.
  /// Useful for initializing default anchors on first launch.
  Future<int> upsertAnchor(MealAnchorsCompanion companion) {
    return into(mealAnchors).insertOnConflictUpdate(companion);
  }
}
```
