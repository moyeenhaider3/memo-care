This is the critical bridge between template definitions (Plan 06-01) and the running app. It orchestrates a 5-step pipeline: create chain → create reminders → create edges → validate DAG → schedule alarms. If validation fails (cycle detected), it rolls back all created records.

The service uses constructor injection for all dependencies (repositories, validator, scheduler), making it fully testable with mocks.

TemplateRepository is deliberately thin — it wraps the in-memory kTemplatePacks constant for API consistency with other features. No network, no database — templates are compile-time constants.

**Template instantiation pipeline:**

```
TemplatePack + userOverrides + mealAnchorTimes
  │
  ├─ 1. chainRepo.createChain(pack.name) → chainId
  │
  ├─ 2. For each TemplateMedicine:
  │      Apply userOverride if exists (name, dosage changes)
  │      Compute scheduledAt from anchorMeal + mealAnchorTimes, or defaultTimeMinutes
  │      reminderRepo.createReminder(...) → reminderId
  │      Store index→reminderId mapping
  │
  ├─ 3. For each TemplateEdge:
  │      Resolve sourceIndex → real sourceReminderId
  │      Resolve targetIndex → real targetReminderId
  │      chainRepo.createEdge(chainId, sourceId, targetId)
  │
  ├─ 4. chainValidator.validate(chainId)
  │      If Left(cycleDetected) → chainRepo.deleteChain(chainId) → return Left
  │
  └─ 5. For each reminder: alarmScheduler.schedule(reminderId, scheduledAt)
        Return Right(TemplateInstantiationResult)
```

/// Template instantiation engine.
///
/// Converts a [TemplatePack] blueprint into real database records:
/// [ReminderChain] + [Reminder]s + [ChainEdge]s.
///
/// Pure Dart — no Flutter imports. All dependencies injected via constructor.

import 'package:fpdart/fpdart.dart';
import 'package:memo_care/features/chain_engine/data/chain_repository.dart';
import 'package:memo_care/features/chain_engine/domain/chain_validator.dart';
import 'package:memo_care/features/onboarding/domain/models/onboarding_state.dart';
import 'package:memo_care/features/reminders/data/reminder_repository.dart';
import 'package:memo_care/features/templates/domain/models/template_pack.dart';
import 'package:memo_care/core/platform/alarm_scheduler.dart';

/// Errors that can occur during template instantiation.
sealed class TemplateError {
  const TemplateError();
}

/// DAG cycle detected in the template's edge structure.
class TemplateCycleDetected extends TemplateError {
  const TemplateCycleDetected();

  @override
  String toString() => 'TemplateCycleDetected: chain contains a cycle';
}

/// A TemplateEdge references an index outside the medicines list.
class TemplateInvalidEdgeIndex extends TemplateError {
  const TemplateInvalidEdgeIndex(this.sourceIndex, this.targetIndex, this.medicineCount);

  final int sourceIndex;
  final int targetIndex;
  final int medicineCount;

  @override
  String toString() =>
      'TemplateInvalidEdgeIndex: edge $sourceIndex→$targetIndex '
      'out of bounds (medicines: $medicineCount)';
}

/// Database or scheduler failure during instantiation.
class TemplatePersistenceFailure extends TemplateError {
  const TemplatePersistenceFailure(this.message);

  final String message;

  @override
  String toString() => 'TemplatePersistenceFailure: $message';
}

/// Result of a successful template instantiation.
class TemplateInstantiationResult {
  const TemplateInstantiationResult({
    required this.chainId,
    required this.reminderIds,
    required this.edgeIds,
  });

  /// The created chain's database ID.
  final int chainId;

  /// Database IDs of the created reminders (ordered by chainPosition).
  final List<int> reminderIds;

  /// Database IDs of the created chain edges.
  final List<int> edgeIds;
}

/// Instantiates [TemplatePack] blueprints into real database records.
///
/// Pipeline:
/// 1. Validate edge indices
/// 2. Create [ReminderChain]
/// 3. Create [Reminder]s (applying user overrides)
/// 4. Create [ChainEdge]s (resolving index → real ID)
/// 5. Run [ChainValidator] (cycle detection)
/// 6. Schedule initial alarms
///
/// If validation fails at step 5, all created records are rolled back.
class TemplateService {
  TemplateService({
    required ChainRepository chainRepository,
    required ReminderRepository reminderRepository,
    required ChainValidator chainValidator,
    required AlarmScheduler alarmScheduler,
  })  : _chainRepo = chainRepository,
        _reminderRepo = reminderRepository,
        _chainValidator = chainValidator,
        _alarmScheduler = alarmScheduler;

  final ChainRepository _chainRepo;
  final ReminderRepository _reminderRepo;
  final ChainValidator _chainValidator;
  final AlarmScheduler _alarmScheduler;

  /// Instantiate a template pack into real database records.
  ///
  /// [pack] — the template blueprint to instantiate.
  /// [userOverrides] — optional medicine customizations keyed by chainPosition.
  ///   If an override exists for a position, its name/dosage/time replace defaults.
  /// [mealAnchorTimes] — meal type → minutes from midnight.
  ///   Used to compute [scheduledAt] for meal-linked medicines.
  ///
  /// Returns [Right(TemplateInstantiationResult)] on success,
  /// or [Left(TemplateError)] on failure.
  Future<Either<TemplateError, TemplateInstantiationResult>> apply({
    required TemplatePack pack,
    Map<int, CustomMedicineEntry> userOverrides = const {},
    Map<String, int> mealAnchorTimes = const {},
  }) async {
    // Step 1: Validate edge indices before touching the database
    for (final edge in pack.edges) {
      if (edge.sourceIndex < 0 ||
          edge.sourceIndex >= pack.medicines.length ||
          edge.targetIndex < 0 ||
          edge.targetIndex >= pack.medicines.length) {
        return left(TemplateInvalidEdgeIndex(
          edge.sourceIndex,
          edge.targetIndex,
          pack.medicines.length,
        ));
      }
    }

    try {
      // Step 2: Create the chain
      final chainId = await _chainRepo.createChain(pack.name);

      // Step 3: Create reminders, collecting index → real ID mapping
      final indexToReminderId = <int, int>{};
      final reminderIds = <int>[];

      for (var i = 0; i < pack.medicines.length; i++) {
        final templateMed = pack.medicines[i];
        final override = userOverrides[templateMed.chainPosition];

        final name = override?.name ?? templateMed.name;
        final dosage = override?.dosage ?? templateMed.defaultDosage;
        final medicineType = override?.medicineType ?? templateMed.medicineType;

        // Compute scheduledAt
        final scheduledAt = _computeScheduledAt(
          templateMed: templateMed,
          override: override,
          mealAnchorTimes: mealAnchorTimes,
        );

        final reminderId = await _reminderRepo.createReminder(
          chainId: chainId,
          name: name,
          medicineType: medicineType,
          dosage: dosage,
          scheduledAt: scheduledAt,
          gapHours: templateMed.gapHours,
        );

        indexToReminderId[i] = reminderId;
        reminderIds.add(reminderId);
      }

      // Step 4: Create edges, resolving indices to real IDs
      final edgeIds = <int>[];
      for (final edge in pack.edges) {
        final sourceId = indexToReminderId[edge.sourceIndex]!;
        final targetId = indexToReminderId[edge.targetIndex]!;
        final edgeId = await _chainRepo.createEdge(
          chainId: chainId,
          sourceId: sourceId,
          targetId: targetId,
        );
        edgeIds.add(edgeId);
      }

      // Step 5: Validate DAG (cycle detection)
      final validationResult = await _chainValidator.validate(chainId);
      if (validationResult.isLeft()) {
        // Rollback: delete the chain (cascades to reminders + edges)
        await _chainRepo.deleteChain(chainId);
        return left(const TemplateCycleDetected());
      }

      // Step 6: Schedule initial alarms for reminders with known times
      for (final entry in indexToReminderId.entries) {
        final templateMed = pack.medicines[entry.key];
        final override = userOverrides[templateMed.chainPosition];
        final scheduledAt = _computeScheduledAt(
          templateMed: templateMed,
          override: override,
          mealAnchorTimes: mealAnchorTimes,
        );
        if (scheduledAt != null) {
          await _alarmScheduler.schedule(entry.value, scheduledAt);
        }
      }

      return right(TemplateInstantiationResult(
        chainId: chainId,
        reminderIds: reminderIds,
        edgeIds: edgeIds,
      ));
    } catch (e) {
      return left(TemplatePersistenceFailure(e.toString()));
    }
  }

  /// Compute the initial scheduled time for a template medicine.
  ///
  /// - For meal-linked meds: use anchor time ± offset from [AnchorConfig] defaults.
  /// - For fixedTime meds: use [defaultTimeMinutes] or override time.
  /// - Returns null if time can't be determined (anchor not set yet).
  DateTime? _computeScheduledAt({
    required TemplateMedicine templateMed,
    CustomMedicineEntry? override,
    required Map<String, int> mealAnchorTimes,
  }) {
    final now = DateTime.now();
    final today = DateTime(now.year, now.month, now.day);

    // Check for override time first
    if (override?.timeMinutes != null) {
      return today.add(Duration(minutes: override!.timeMinutes!));
    }

    // Fixed-time medicines use their default time
    if (templateMed.defaultTimeMinutes != null) {
      return today.add(Duration(minutes: templateMed.defaultTimeMinutes!));
    }

    // Meal-linked medicines: compute from anchor time + offset
    if (templateMed.anchorMeal != null) {
      final mealKey = templateMed.anchorMeal!.name; // e.g., 'breakfast'
      final anchorMinutes = mealAnchorTimes[mealKey];
      if (anchorMinutes != null) {
        final anchorTime = today.add(Duration(minutes: anchorMinutes));
        return switch (templateMed.medicineType) {
          MedicineType.beforeMeal => anchorTime.subtract(const Duration(minutes: 30)),
          MedicineType.afterMeal => anchorTime.add(const Duration(minutes: 30)),
          _ => anchorTime,
        };
      }
    }

    return null; // Time will be resolved later by AnchorResolver
  }
}

/// In-memory template repository.
///
/// Templates are compile-time constants — no database, no network.
/// This class exists for API consistency with other features.

import 'package:memo_care/features/templates/domain/models/template_pack.dart';
import 'package:memo_care/features/templates/domain/models/template_packs.dart';

/// Provides access to the pre-defined template packs.
class TemplateRepository {
  const TemplateRepository();

  /// Returns all available template packs.
  List<TemplatePack> getAll() => kTemplatePacks;

  /// Returns the template pack with the given [id], or null if not found.
  TemplatePack? getById(String id) {
    for (final pack in kTemplatePacks) {
      if (pack.id == id) return pack;
    }
    return null;
  }

  /// Returns template packs matching the given [condition].
  List<TemplatePack> getByCondition(String condition) {
    return kTemplatePacks.where((p) => p.condition == condition).toList();
  }
}
