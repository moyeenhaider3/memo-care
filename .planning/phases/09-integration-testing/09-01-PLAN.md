extension CommonActions on PatrolIntegrationTester {
  /// Complete the full onboarding flow with default selections.
  /// Returns after landing on the home screen.
  Future<void> completeOnboarding({
    String condition = 'Diabetes',
    String template = 'Diabetic Pack',
  }) async {
    // Step 1: Select condition
    await $(condition).tap();
    await $('Next').tap();

    // Step 2: Select template
    await $(template).tap();
    await $('Next').tap();

    // Step 3: Set meal anchors (accept defaults)
    await $('Next').tap();

    // Step 4: Grant permissions
    // Use native interaction to grant Android permissions
    await $.native.grantPermissionWhenInUse();

    // Step 5: Review and confirm
    await $('Done').tap();

    // Wait for home screen
    await $.pumpAndSettle();
  }

  /// Confirm a reminder by tapping DONE on its notification.
  Future<void> confirmReminderFromNotification(String medicineName) async {
    // Open notification shade
    await $.native.openNotifications();

    // Find and tap the DONE action on the notification
    await $.native.tap(Selector(textContains: medicineName));
    await $.native.tap(Selector(text: 'DONE'));
  }

  /// Snooze a reminder from notification.
  Future<void> snoozeReminderFromNotification(String medicineName) async {
    await $.native.openNotifications();
    await $.native.tap(Selector(textContains: medicineName));
    await $.native.tap(Selector(text: 'SNOOZE'));
  }

  /// Skip a reminder from notification.
  Future<void> skipReminderFromNotification(String medicineName) async {
    await $.native.openNotifications();
    await $.native.tap(Selector(textContains: medicineName));
    await $.native.tap(Selector(text: 'SKIP'));
  }

  /// Navigate to a screen via bottom navigation or drawer.
  Future<void> navigateTo(String screenName) async {
    await $(screenName).tap();
    await $.pumpAndSettle();
  }

  /// Wait for a notification to appear with the given text.
  Future<void> waitForNotification(String text, {Duration timeout = const Duration(seconds: 30)}) async {
    await $.native.openNotifications();
    // Poll for notification appearance
    final end = DateTime.now().add(timeout);
    while (DateTime.now().isBefore(end)) {
      try {
        await $.native.tap(Selector(textContains: text));
        return;
      } catch (_) {
        await Future<void>.delayed(const Duration(seconds: 2));
        await $.native.openNotifications();
      }
    }
    throw TimeoutException('Notification with "$text" did not appear within $timeout');
  }
}
