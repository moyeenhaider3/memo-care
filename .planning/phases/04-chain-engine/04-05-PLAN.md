part 'chain_notifier.g.dart';

/// State for a single chain — holds the chain metadata, its reminders, and edges.
class ChainState {
  const ChainState({
    required this.chain,
    required this.reminders,
    required this.edges,
  });

  final ReminderChain chain;
  final List<Reminder> reminders;
  final List<ChainEdge> edges;
}

/// Manages the state of a single reminder chain.
///
/// Loads chain data reactively and provides mutation methods
/// that validate via [ChainValidator] before persisting.
@riverpod
class ChainNotifier extends _$ChainNotifier {
  @override
  Future<ChainState> build(int chainId) async {
    final repo = ref.watch(chainRepositoryProvider);
    final chain = await repo.getById(chainId);
    final reminders = await repo.getReminders(chainId);
    final edges = await repo.getEdges(chainId);

    return ChainState(
      chain: chain!,
      reminders: reminders,
      edges: edges,
    );
  }

  /// Adds an edge to the chain after validating no cycles are introduced.
  ///
  /// Returns `Right(void)` on success, `Left(ChainError)` on validation failure.
  Future<Either<ChainError, void>> addEdge({
    required int sourceId,
    required int targetId,
  }) async {
    final currentState = await future;
    final validator = ref.read(chainValidatorProvider);
    final repo = ref.read(chainRepositoryProvider);

    // Tentatively add edge and validate
    final newEdges = [
      ...currentState.edges.map((e) => (sourceId: e.sourceId, targetId: e.targetId)),
      (sourceId: sourceId, targetId: targetId),
    ];
    final nodeIds = currentState.reminders.map((r) => r.id).toList();

    final validation = validator.validate(
      nodeIds: nodeIds,
      edges: newEdges,
    );

    return validation.fold(
      (error) => left(error),
      (_) async {
        await repo.addEdge(
          chainId: currentState.chain.id,
          sourceId: sourceId,
          targetId: targetId,
        );
        ref.invalidateSelf();
        return right(null);
      },
    );
  }

  /// Removes an edge from the chain.
  Future<void> removeEdge(int edgeId) async {
    final repo = ref.read(chainRepositoryProvider);
    await repo.removeEdge(edgeId);
    ref.invalidateSelf();
  }
}
```

import 'package:riverpod_annotation/riverpod_annotation.dart';

import 'package:memo_care/features/confirmation/application/providers.dart';
import 'package:memo_care/features/confirmation/domain/confirmation_service.dart';
import 'package:memo_care/features/confirmation/domain/models/confirmation_state.dart';
import 'package:memo_care/features/reminders/domain/models/reminder.dart';
import 'package:memo_care/core/platform/alarm_scheduler.dart';

part 'confirmation_notifier.g.dart';

/// Handles user confirmation actions (DONE/SNOOZE/SKIP) end-to-end.
///
/// Flow:
/// 1. Calls [ConfirmationService.confirm()] for domain logic + persistence
/// 2. Dispatches [AlarmScheduler] side effects based on [ConfirmationOutcome]
/// 3. Updates [ReminderRepository] for is_active status changes
/// 4. Invalidates self to refresh UI
@riverpod
class ConfirmationNotifier extends _$ConfirmationNotifier {
  @override
  FutureOr<void> build() {
    // No initial state — this notifier is action-driven
  }

  /// Confirms a reminder with the given [state].
  ///
  /// [snoozeUntil] is required when [state] is [ConfirmationState.snoozed].
  /// After processing, triggers UI refresh via provider invalidation.
  Future<void> confirm({
    required int reminderId,
    required int chainId,
    required ConfirmationState state,
    DateTime? snoozeUntil,
  }) async {
    state_ = const AsyncLoading();

    try {
      final service = ref.read(confirmationServiceProvider);
      final scheduler = ref.read(alarmSchedulerProvider);

      final outcome = await service.confirm(
        reminderId: reminderId,
        chainId: chainId,
        state: state,
        snoozeUntil: snoozeUntil,
      );

      switch (outcome) {
        case ActivateDownstream(:final reminders):
          // Schedule alarms for immediate downstream reminders
          for (final reminder in reminders) {
            if (reminder.scheduledAt != null) {
              await scheduler.schedule(reminder);
            }
          }
          // Mark downstream reminders as active
          await _activateReminders(reminders);

        case SuspendDownstream(:final reminders):
          // Cancel alarms for all suspended reminders
          for (final reminder in reminders) {
            await scheduler.cancel(reminder.id);
          }
          // Mark suspended reminders as inactive
          await _deactivateReminders(reminders);

        case RescheduleSnooze(:final reminder, :final remainingSnoozes):
          // Reschedule the same reminder at snooze time
          if (reminder != null && snoozeUntil != null) {
            final updatedReminder = (reminder as Reminder).copyWith(
              scheduledAt: snoozeUntil,
            );
            await scheduler.schedule(updatedReminder);
          }

        case AutoSkipped(:final reason, :final suspendedReminders):
          // Cancel alarms for all downstream (same as SKIPPED)
          for (final reminder in suspendedReminders) {
            await scheduler.cancel((reminder as Reminder).id);
          }
          await _deactivateReminders(suspendedReminders.cast<Reminder>());
          // TODO: Show user notification about auto-skip reason

        case ConfirmationFailed(:final error):
          state_ = AsyncError(error, StackTrace.current);
          return;
      }

      // Refresh chain state for any watchers
      ref.invalidate(chainNotifierProvider(chainId));
      state_ = const AsyncData(null);

    } catch (e, st) {
      state_ = AsyncError(e, st);
    }
  }

  /// Marks reminders as active (is_active = true) in the database.
  Future<void> _activateReminders(List<dynamic> reminders) async {
    final repo = ref.read(reminderRepositoryProvider);
    for (final reminder in reminders) {
      final r = reminder as Reminder;
      await repo.update(r.copyWith(isActive: true));
    }
  }

  /// Marks reminders as inactive (is_active = false) in the database.
  Future<void> _deactivateReminders(List<dynamic> reminders) async {
    final repo = ref.read(reminderRepositoryProvider);
    for (final reminder in reminders) {
      final r = reminder as Reminder;
      await repo.update(r.copyWith(isActive: false));
    }
  }
}
```
