**DB schema for queries:**

Today's reminders query:

```sql
SELECT r.* FROM reminders r
WHERE r.scheduled_at >= :todayStartUtc
  AND r.scheduled_at < :tomorrowStartUtc
  AND r.is_active = 1
ORDER BY r.scheduled_at ASC;
```

Missed reminders query:

```sql
SELECT r.* FROM reminders r
LEFT JOIN confirmations c ON c.reminder_id = r.id
WHERE r.scheduled_at < :nowUtc
  AND r.is_active = 1
  AND (c.id IS NULL OR c.state = 'snoozed')
ORDER BY r.scheduled_at ASC;
```

Note: "missed" = scheduled in the past AND no DONE/SKIPPED confirmation. Snoozed counts as still pending.

History page query:

```sql
SELECT r.*, c.state as confirmation_state, c.confirmed_at
FROM reminders r
LEFT JOIN confirmations c ON c.reminder_id = r.id AND c.state != 'snoozed'
WHERE (:filter IS NULL OR r.medicine_name LIKE '%' || :filter || '%')
ORDER BY r.scheduled_at DESC
LIMIT :limit OFFSET :offset;
```

Distinct medicine names:

```sql
SELECT DISTINCT medicine_name FROM reminders ORDER BY medicine_name ASC;
```

/// Watches today's reminders in chronological order (VIEW-01).
///
/// Filters to reminders with scheduled_at between [todayStart] and [todayEnd]
/// in UTC epoch millis. Auto-emits on any change to the reminders table.
Stream<List<ReminderRow>> watchTodayReminders({
  required int todayStartUtcMs,
  required int todayEndUtcMs,
}) {
  return (select(reminders)
    ..where((r) => r.scheduledAt.isBiggerOrEqualValue(todayStartUtcMs))
    ..where((r) => r.scheduledAt.isSmallerThanValue(todayEndUtcMs))
    ..where((r) => r.isActive.equals(true))
    ..orderBy([(r) => OrderingTerm.asc(r.scheduledAt)]))
      .watch();
}

/// Watches reminders that are missed — scheduled in the past and still pending (VIEW-04).
///
/// A reminder is "missed" if scheduled_at < now AND no DONE/SKIPPED confirmation exists.
/// Uses a custom query joining with confirmations table.
Stream<List<ReminderRow>> watchMissedReminders({
  required int nowUtcMs,
}) {
  // Reminders scheduled before now, active, with no terminal confirmation
  final query = customSelect(
    'SELECT r.* FROM reminders r '
    'LEFT JOIN confirmations c ON c.reminder_id = r.id AND c.state IN (\'done\', \'skipped\') '
    'WHERE r.scheduled_at < ? AND r.is_active = 1 AND c.id IS NULL '
    'ORDER BY r.scheduled_at ASC',
    variables: [Variable.withInt(nowUtcMs)],
    readsFrom: {reminders, confirmations},
  );
  return query.watch().map(
    (rows) => rows.map((row) => reminders.map(row.data)).toList(),
  );
}

/// Fetches a page of history entries with optional medicine name filter (HIST-01, HIST-02, HIST-03).
///
/// Returns reminder rows joined with their latest terminal confirmation (done/skipped).
/// Ordered by scheduled_at DESC (most recent first).
/// [limit] and [offset] for pagination; [medicineNameFilter] for LIKE search.
Future<List<HistoryQueryRow>> getHistoryPage({
  required int limit,
  required int offset,
  String? medicineNameFilter,
}) {
  final filterClause = medicineNameFilter != null
      ? 'AND r.medicine_name LIKE ?'
      : '';
  final variables = <Variable>[
    if (medicineNameFilter != null) Variable.withString('%$medicineNameFilter%'),
    Variable.withInt(limit),
    Variable.withInt(offset),
  ];

  return customSelect(
    'SELECT r.id AS r_id, r.medicine_name, r.dosage, r.scheduled_at, '
    'c.state AS confirmation_state, c.confirmed_at '
    'FROM reminders r '
    'LEFT JOIN confirmations c ON c.reminder_id = r.id AND c.state IN (\'done\', \'skipped\') '
    'WHERE 1=1 $filterClause '
    'ORDER BY r.scheduled_at DESC '
    'LIMIT ? OFFSET ?',
    variables: variables,
    readsFrom: {reminders, confirmations},
  ).get().then(
    (rows) => rows.map((row) => HistoryQueryRow(
      reminderId: row.read<int>('r_id'),
      medicineName: row.read<String>('medicine_name'),
      dosage: row.readNullable<String>('dosage'),
      scheduledAt: row.read<int>('scheduled_at'),
      confirmationState: row.readNullable<String>('confirmation_state'),
      confirmedAt: row.readNullable<int>('confirmed_at'),
    )).toList(),
  );
}

/// Returns total count of history entries matching filter — for pagination UI.
Future<int> getHistoryTotalCount({String? medicineNameFilter}) {
  final filterClause = medicineNameFilter != null
      ? 'AND r.medicine_name LIKE ?'
      : '';
  final variables = <Variable>[
    if (medicineNameFilter != null) Variable.withString('%$medicineNameFilter%'),
  ];

  return customSelect(
    'SELECT COUNT(*) AS cnt FROM reminders r WHERE 1=1 $filterClause',
    variables: variables,
    readsFrom: {reminders},
  ).getSingle().then((row) => row.read<int>('cnt'));
}

/// Returns distinct medicine names for the history filter dropdown (HIST-03).
Future<List<String>> getDistinctMedicineNames() {
  return customSelect(
    'SELECT DISTINCT medicine_name FROM reminders ORDER BY medicine_name ASC',
    readsFrom: {reminders},
  ).get().then(
    (rows) => rows.map((row) => row.read<String>('medicine_name')).toList(),
  );
}

class HistoryEntry with _$HistoryEntry {

    required int reminderId,

    String? dosage,
    required DateTime scheduledAt,
    ConfirmationState? status,
    DateTime? confirmedAt,

      medicineName: row.medicineName,

    ConfirmationState.done => 'Taken',
    ConfirmationState.skipped => 'Skipped',
    ConfirmationState.snoozed => 'Snoozed',

  };

/// State for today's schedule view (VIEW-01, VIEW-02, VIEW-04).
@freezed
class DailyScheduleState with _$DailyScheduleState {
  const factory DailyScheduleState({
    /// All of today's reminders in chronological order.
    required List<Reminder> todayReminders,

    /// The next pending reminder (scheduled in the future, still pending).
    /// Shown as the hero card (VIEW-02).
    Reminder? nextPending,

    /// Reminders scheduled in the past with no terminal confirmation (VIEW-04).
    required List<Reminder> missedReminders,
  }) = _DailyScheduleState;
}

/// Manages the daily schedule state by watching today's reminders from Drift.
///
/// Automatically recomputes [nextPending] and [missedReminders] whenever
/// the reminders table changes (e.g., a confirmation is recorded).
///
/// Split from a single giant provider to avoid rebuild avalanche (PITFALLS.md):
/// - This notifier manages the full state
/// - Derived providers select specific slices (next pending only, missed only)
@riverpod
class DailyScheduleNotifier extends _$DailyScheduleNotifier {
  StreamSubscription<List<Reminder>>? _todaySub;
  StreamSubscription<List<Reminder>>? _missedSub;

  @override
  Future<DailyScheduleState> build() async {
    final repo = ref.watch(reminderRepositoryProvider);

    // Watch today's reminders
    final todayStream = repo.watchToday();
    final missedStream = repo.watchMissed();

    // Get initial data
    final todayReminders = await todayStream.first;
    final missedReminders = await missedStream.first;

    // Set up reactive subscriptions
    _todaySub?.cancel();
    _todaySub = todayStream.listen((reminders) {
      _updateState(todayReminders: reminders);
    });

    _missedSub?.cancel();
    _missedSub = missedStream.listen((missed) {
      _updateState(missedReminders: missed);
    });

    ref.onDispose(() {
      _todaySub?.cancel();
      _missedSub?.cancel();
    });

    return _computeState(todayReminders, missedReminders);
  }

  void _updateState({
    List<Reminder>? todayReminders,
    List<Reminder>? missedReminders,
  }) {
    final current = state.valueOrNull;
    if (current == null) return;

    final today = todayReminders ?? current.todayReminders;
    final missed = missedReminders ?? current.missedReminders;
    state = AsyncData(_computeState(today, missed));
  }

  DailyScheduleState _computeState(
    List<Reminder> todayReminders,
    List<Reminder> missedReminders,
  ) {
    final now = DateTime.now().toUtc();

    // Next pending = first reminder that is scheduled in the future and still active
    // (not yet confirmed as DONE or SKIPPED)
    // The "missed" list already filters past+pending, so nextPending is future+pending
    final pendingFuture = todayReminders.where((r) {
      final scheduledAt = r.scheduledAt;
      return scheduledAt != null && scheduledAt.isAfter(now);
    }).toList();

    final nextPending = pendingFuture.isNotEmpty ? pendingFuture.first : null;

    return DailyScheduleState(
      todayReminders: todayReminders,
      nextPending: nextPending,
      missedReminders: missedReminders,
    );
  }
}

/// Manages paginated medication history with optional filter (HIST-02, HIST-03).
///
/// Page size is 20 to avoid loading 1000+ rows (PITFALLS.md — Drift query in UI thread).
/// [loadNextPage] appends results; [setFilter] resets and reloads.
@riverpod
class HistoryNotifier extends _$HistoryNotifier {
  static const _pageSize = 20;

  @override
  Future<HistoryState> build() async {
    final repo = ref.watch(reminderRepositoryProvider);

    // Load available filter options
    final filters = await repo.getDistinctMedicineNames();

    // Load first page
    final items = await repo.getHistoryPage(
      limit: _pageSize,
      offset: 0,
    );

    return HistoryState(
      items: items,
      hasMore: items.length >= _pageSize,
      currentPage: 0,
      availableFilters: filters,
    );
  }

  /// Loads the next page of history entries.
  ///
  /// No-op if already loading or no more pages available.
  Future<void> loadNextPage() async {
    final current = state.valueOrNull;
    if (current == null || !current.hasMore || current.isLoadingMore) return;

    state = AsyncData(current.copyWith(isLoadingMore: true));

    final repo = ref.read(reminderRepositoryProvider);
    final nextPage = current.currentPage + 1;
    final offset = nextPage * _pageSize;

    final newItems = await repo.getHistoryPage(
      limit: _pageSize,
      offset: offset,
      medicineNameFilter: current.activeFilter,
    );

    state = AsyncData(current.copyWith(
      items: [...current.items, ...newItems],
      hasMore: newItems.length >= _pageSize,
      currentPage: nextPage,
      isLoadingMore: false,
    ));
  }

  /// Sets the medication name filter and reloads from page 0.
  ///
  /// Pass null to clear the filter.
  Future<void> setFilter(String? medicineFilter) async {
    state = const AsyncLoading();

    final repo = ref.read(reminderRepositoryProvider);
    final items = await repo.getHistoryPage(
      limit: _pageSize,
      offset: 0,
      medicineNameFilter: medicineFilter,
    );

    final filters = await repo.getDistinctMedicineNames();

    state = AsyncData(HistoryState(
      items: items,
      hasMore: items.length >= _pageSize,
      currentPage: 0,
      activeFilter: medicineFilter,
      availableFilters: filters,
    ));
  }

  /// Refreshes from page 0 with current filter.
  Future<void> refresh() async {
    final currentFilter = state.valueOrNull?.activeFilter;
    await setFilter(currentFilter);
  }
}
