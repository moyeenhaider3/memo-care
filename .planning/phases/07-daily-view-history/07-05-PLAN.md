<objective>
Build the chain context detail view — when a user taps any reminder on the home screen, they see what triggered it and what it will trigger next (VIEW-03).

This is the key differentiator UI: it makes the DAG-based chain structure visible to the user. A patient can see "I took my insulin, which triggered the metformin reminder, which will trigger the afternoon check."

The screen uses a vertical timeline layout with upstream nodes above, the current reminder in the center, and downstream nodes below. Each node shows medicine name, status badge, and connection lines.

chainContextProvider queries ChainDao for edges and resolves the connected reminders. This is a FutureProvider (not stream) since chain structure changes infrequently — only on chain mutation.

The route is placed OUTSIDE the ShellRoute because it's a detail/drill-down screen, not a tab destination. It has a back button to return to the home screen.

Output: Chain context screen + provider + route + unit test.
</objective>

**DB schema for chain context queries:**

```sql
-- Find chain_id for a reminder
SELECT chain_id FROM reminders WHERE id = :reminderId;

-- Find downstream (what this reminder triggers)
SELECT e.target_id, r.*
FROM chain_edges e
JOIN reminders r ON r.id = e.target_id
WHERE e.source_id = :reminderId;

-- Find upstream (what triggered this reminder)
SELECT e.source_id, r.*
FROM chain_edges e
JOIN reminders r ON r.id = e.source_id
WHERE e.target_id = :reminderId;

-- Chain name
SELECT name FROM reminder_chains WHERE id = :chainId;
```

```dart
import 'package:freezed_annotation/freezed_annotation.dart';
import 'package:riverpod_annotation/riverpod_annotation.dart';

import 'package:memo_care/features/chain_engine/data/chain_dao.dart';
import 'package:memo_care/features/reminders/data/reminder_dao.dart';
import 'package:memo_care/features/reminders/domain/models/reminder.dart';

part 'chain_context_providers.freezed.dart';
part 'chain_context_providers.g.dart';

/// The resolved chain context for a single reminder (VIEW-03).
///
/// Contains the current reminder, its upstream parents (what triggered it),
/// and downstream children (what it triggers).
@freezed
class ChainContext with _$ChainContext {
  const factory ChainContext({
    /// The reminder being inspected.
    required Reminder currentReminder,

    /// Reminders that trigger this one (upstream edges where target == current).
    required List<Reminder> upstreamReminders,

    /// Reminders that this one triggers (downstream edges where source == current).
    required List<Reminder> downstreamReminders,

    /// Name of the chain this reminder belongs to.
    required String chainName,
  }) = _ChainContext;
}

/// Loads the chain context for a given reminder ID.
///
/// Queries:
/// 1. Load the reminder to get its chain_id
/// 2. Load the chain name
/// 3. Find downstream edges (source_id == reminderId) → load target reminders
/// 4. Find upstream edges (target_id == reminderId) → load source reminders
///
/// FutureProvider (not stream) because chain structure changes infrequently.
@riverpod
Future<ChainContext> chainContext(Ref ref, int reminderId) async {
  final reminderDao = ref.watch(reminderDaoProvider);
  final chainDao = ref.watch(chainDaoProvider);

  // 1. Load current reminder
  final currentRow = await reminderDao.getReminderById(reminderId);
  if (currentRow == null) {
    throw Exception('Reminder $reminderId not found');
  }
  final current = Reminder.fromRow(currentRow);

  // 2. Load chain name
  final chainId = current.chainId;
  final chain = await chainDao.getChainById(chainId);
  final chainName = chain?.name ?? 'Unknown Chain';

  // 3. Find downstream: edges where source_id == reminderId
  final downstreamEdges = await chainDao.getEdgesFromSource(reminderId);
  final downstreamReminders = <Reminder>[];
  for (final edge in downstreamEdges) {
    final row = await reminderDao.getReminderById(edge.targetId);
    if (row != null) {
      downstreamReminders.add(Reminder.fromRow(row));
    }
  }

  // 4. Find upstream: edges where target_id == reminderId
  final upstreamEdges = await chainDao.getEdgesToTarget(reminderId);
  final upstreamReminders = <Reminder>[];
  for (final edge in upstreamEdges) {
    final row = await reminderDao.getReminderById(edge.sourceId);
    if (row != null) {
      upstreamReminders.add(Reminder.fromRow(row));
    }
  }

  return ChainContext(
    currentReminder: current,
    upstreamReminders: upstreamReminders,
    downstreamReminders: downstreamReminders,
    chainName: chainName,
  );
}
```

```dart
class ChainContextScreen extends ConsumerWidget {
  const ChainContextScreen({
    required this.reminderId,
    super.key,
  });

  final int reminderId;

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final contextAsync = ref.watch(chainContextProvider(reminderId));
    final theme = Theme.of(context);

    return Scaffold(
      appBar: AppBar(
        leading: IconButton(
          icon: const Icon(Icons.arrow_back, size: 28),

/// A single node in the chain visualization.
class _ChainNode extends StatelessWidget {
  const _ChainNode({
    required this.reminder,
    required this.isCurrent,
    required this.showConnector,
  });

  final Reminder reminder;
  final bool isCurrent;
  final bool showConnector;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);
    final timeFormat = DateFormat.jm();
    final timeText = reminder.scheduledAt != null
        ? timeFormat.format(reminder.scheduledAt!.toLocal())
        : '--:--';

    return Semantics(
      label: '${isCurrent ? "Current" : "Chain"} reminder: '
          '${reminder.medicineName}, '
          '${reminder.dosage ?? ""}, '
          'at $timeText',
      button: !isCurrent,
      child: GestureDetector(
        onTap: isCurrent
            ? null
            : () => context.push('/reminder/${reminder.id}/chain'),
        child: Container(
          margin: const EdgeInsets.only(bottom: 8),
          padding: const EdgeInsets.all(16),
          constraints: const BoxConstraints(minHeight: 56),
          decoration: BoxDecoration(
            color: isCurrent
                ? theme.colorScheme.primaryContainer
                : theme.colorScheme.surfaceContainerHighest,
            borderRadius: BorderRadius.circular(12),
            border: isCurrent
                ? Border.all(color: theme.colorScheme.primary, width: 2)
                : null,
          ),
          child: Row(
            children: [
              // Node indicator
              Container(
                width: 12,
                height: 12,
                decoration: BoxDecoration(
                  shape: BoxShape.circle,
                  color: isCurrent
                      ? theme.colorScheme.primary
                      : theme.colorScheme.outline,
                ),
              ),
              const SizedBox(width: 12),

              // Medicine info
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text(
                      reminder.medicineName,
                      style: theme.textTheme.bodyLarge?.copyWith(
                        fontWeight: isCurrent ? FontWeight.bold : FontWeight.w600,
                        color: isCurrent
                            ? theme.colorScheme.onPrimaryContainer
                            : null,
                      ),
                    ),
                    Text(
                      [
                        if (reminder.dosage != null) reminder.dosage!,
                        timeText,
                      ].join(' · '),
                      style: theme.textTheme.bodyMedium?.copyWith(
                        color: isCurrent
                            ? theme.colorScheme.onPrimaryContainer.withOpacity(0.8)
                            : theme.colorScheme.onSurfaceVariant,
                      ),
                    ),
                  ],
                ),
              ),

              // Status badge
              StatusBadge(
                status: null, // TODO: needs confirmation status lookup
                isMissed: _isMissed,
              ),

              // Navigate arrow for non-current nodes
              if (!isCurrent) ...[
                const SizedBox(width: 8),
                Icon(
                  Icons.chevron_right,
                  color: theme.colorScheme.onSurfaceVariant,
                ),
              ],
            ],
          ),
        ),
      ),
    );
  }

  bool get _isMissed {
    final scheduledAt = reminder.scheduledAt;
    if (scheduledAt == null) return false;
    return scheduledAt.isBefore(DateTime.now().toUtc());
  }
}
```
